```javascript
import React, { useState } from 'react';
import { useFirebaseApp } from 'react-firebase-hooks';
import { getFirestore, doc, setDoc } from 'firebase/firestore'; // שימוש בfirebase/firestore
import { auth } from './firebase'; // חשוב לייבא את auth

function ChatForm() {
  const [message, setMessage] = useState('');
  const firestore = useFirebaseApp().firestore(); // שימוש בfirebaseApp

  const sendMessage = async () => {
    if (!message) return;

    const db = getFirestore(); // שימוש בgetFirestore
    await setDoc(doc(db, 'messages', new Date().getTime()), {
      text: message,
      author: auth.currentUser?.email || 'Anonymous',
      timestamp: new Date().getTime(),
    });

    setMessage('');
  };

  return (
    <div className="input-group">
      <input
        type="text"
        className="form-control"
        placeholder="Send message..."
        value={message}
        onChange={(e) => setMessage(e.target.value)}
      />
      <button className="btn btn-primary" onClick={sendMessage}>Send</button>
    </div>
  );
}

export default ChatForm;
```

**4.3 ChatMessages.js:**

```javascript
import React, { useState, useEffect } from 'react';
import { useFirebaseApp } from 'react-firebase-hooks';
import { getFirestore, collection, query, orderBy, onSnapshot } from 'firebase/firestore'; // שימוש בfirebase/firestore

function ChatMessages() {
  const [messages, setMessages] = useState([]);
  const firestore = useFirebaseApp().firestore(); // שימוש בfirebaseApp

  useEffect(() => {
    const db = getFirestore(); // שימוש בgetFirestore
    const collectionRef = collection(db, 'messages');
    const q = query(collectionRef, orderBy('timestamp'));

    const unsubscribe = onSnapshot(q, (snapshot) => {
      const newMessageArray = snapshot.docs.map((doc) => ({
        id: doc.id,
        ...doc.data(),
      }));
      setMessages(newMessageArray);
    });

    return () => unsubscribe(); // ניקוי המנוי בעת פירוק הרכיב
  }, []);

  return (
    <div>
      {messages.map((message) => (
        <div key={message.id} className="message">
          <p>{message.author}: {message.text}</p>
        </div>
      ))}
    </div>
  );
}

export default ChatMessages;
```

## רגולריזציה בלמידת מכונה: מדריך מפורט

רגולריזציה היא טכניקה חשובה בלמידת מכונה שמטרתה למנוע overfitting - מצב בו מודל לומד את הנתונים בצורה מושלמת מדי, עד כדי שהוא לא מצליח להכליל נתונים חדשים. רגולריזציה מוסיפה מידע נוסף לפונקציית העלות של המודל, שמגדירה את העונש על מודלים מורכבים מדי. הנה סקירה מפורטת של סוגי הרגולריזציה השונים:

**1. L1 Regularization (Lasso):**

*   **הגדרה:** מוסיפה לפונקציית העלות את סכום הערכים המוחלטים של המשקלים של המודל.
*   **נוסחה:** `Cost = Loss + λ * Σ |wᵢ|`, כאשר `Loss` היא פונקציית העלות הרגילה, `λ` הוא פרמטר הרגולריזציה, ו-`wᵢ` הם המשקלים של המודל.
*   **יתרונות:**
    *   **Feature Selection:** נוטה לגרום למשקלים של תכונות לא רלוונטיות להיות שווים לאפס, ובכך לבצע בחירת תכונות באופן אוטומטי.
    *   **מודלים פשוטים:** עוזרת ליצור מודלים פשוטים יותר וקלפירישים יותר.
*   **חסרונות:**
    *   **לא דיפרנציאבילית בנקודה האפס:** עלולה לגרום לבעיות באלגוריתמי אופטימיזציה מסוימים.
*   **דוגמת קוד (Python):**

```python
import numpy as np
from sklearn.linear_model import Lasso
from sklearn.datasets import make_regression

# יצירת נתונים לדוגמה
X, y = make_regression(n_samples=100, n_features=10, random_state=42)

# יצירת מודל Lasso
lasso = Lasso(alpha=1.0) # alpha הוא פרמטר הרגולריזציה λ

# אימון המודל
lasso.fit(X, y)

# הצגת המשקלים של המודל
print(lasso.coef_)
```

**2. L2 Regularization (Ridge):**

*   **הגדרה:** מוסיפה לפונקציית העלות את סכום הריבועים של המשקלים של המודל.
*   **נוסחה:** `Cost = Loss + λ * Σ wᵢ²`, כאשר `Loss` היא פונקציית העלות הרגילה, `λ` הוא פרמטר הרגולריזציה, ו-`wᵢ` הם המשקלים של המודל.
*   **יתרונות:**
    *   **פשוטה ודיפרנציאבילית:** קלה ליישום ומתאימה לאלגוריתמי אופטימיזציה רבים.
    *   **יציבות:** משפרת את יציבות המודל ומונעת משקלים גדולים מדי.
*   **חסרונות:**
    *   **לא מבצעת Feature Selection:** לא גורמת למשקלים של תכונות לא רלוונטיות להיות שווים לאפס.
*   **דוגמת קוד (Python):**

```python
import numpy as np
from sklearn.linear_model import Ridge
from sklearn.datasets import make_regression

# יצירת נתונים לדוגמה
X, y = make_regression(n_samples=100, n_features=10, random_state=42)

# יצירת מודל Ridge
ridge = Ridge(alpha=1.0) # alpha הוא פרמטר הרגולריזציה λ

# אימון המודל
ridge.fit(X, y)

# הצגת המשקלים של המודל
print(ridge.coef_)
```

**3. Elastic Net Regularization:**

*   **הגדרה:** שילוב של L1 ו-L2 רגולריזציה.
*   **נוסחה:** `Cost = Loss + λ₁ * Σ |wᵢ| + λ₂ * Σ wᵢ²`, כאשר `Loss` היא פונקציית העלות הרגילה, `λ₁` ו-`λ₂` הם פרמטרי הרגולריזציה.
*   **יתרונות:**
    *   **משלבת את היתרונות של L1 ו-L2:** מבצעת בחירת תכונות ויציבות.
    *   **גמישה:** מאפשרת לשלוט באיזון בין L1 ו-L2 רגולריזציה.
*   **חסרונות:**
    *   **דורשת כוונון של שני פרמטרים:** `λ₁` ו-`λ₂`.
*   **דוגמת קוד (Python):**

```python
import numpy as np
from sklearn.linear_model import ElasticNet
from sklearn.datasets import make_regression

# יצירת נתונים לדוגמה
X, y = make_regression(n_samples=100, n_features=10, random_state=42)

# יצירת מודל ElasticNet
elastic_net = ElasticNet(alpha=1.0, l1_ratio=0.5) # alpha הוא פרמטר הרגולריזציה λ, l1_ratio הוא היחס בין L1 ל-L2

# אימון המודל
elastic_net.fit(X, y)

# הצגת המשקלים של המודל
print(elastic_net.coef_)
```

*   **יתרונות:**
    *   **מניעת overfitting:** עוזרת למנוע overfitting ויכולה לשפר את הביצועים של הרשת על נתונים חדשים.
    *   **פשוטה ליישום:** קלה ליישום ברשתות נוירונים.
*   **חסרונות:**
