```javascript
// components/ChatForm.jsx
import React, { useState } from 'react';
import { firebase } from '../firebase';

function ChatForm({ setUser }) {
  const [message, setMessage] = useState('');

  const handleSubmit = async () => {
    if (message) {
      try {
        const user = firebase.auth().currentUser;
        if (user) {
          await firebase.firestore().collection('chats').add({
            text: message,
            uid: user.uid,
            createdAt: firebase.firestore.FieldValue.serverTimestamp()
          });
          setMessage('');
        } else {
          console.log("User not authenticated"); // Add this line for debugging
        }
      } catch (error) {
        console.error('Error adding message:', error);
      }
    }
  };

  return (
    <div>
      <input
        type="text"
        value={message}
        onChange={(e) => setMessage(e.target.value)}
      />
      <button onClick={handleSubmit}>Send</button>
    </div>
  );
}

export default ChatForm;

// components/ChatMessages.jsx
import React, { useEffect, useState } from 'react';
import { firebase } from '../firebase';

function ChatMessages() {
  const [messages, setMessages] = useState([]);

  useEffect(() => {
    const unsubscribe = firebase.firestore()
      .collection('chats')
      .orderBy('createdAt', 'asc')
      .onSnapshot((snapshot) => {
        const newMessages = snapshot.docs.map((doc) => ({
          id: doc.id,
          ...doc.data()
        }));
        setMessages(newMessages);
      });

    return () => unsubscribe();
  }, []);

  return (
    <div>
      {messages.map((message) => (
        <div key={message.id}>
          <p>{message.text}</p>
        </div>
      ))}
    </div>
  );
}

export default ChatMessages;

// components/LoginForm.jsx
import React, { useState } from 'react';
import { firebase } from '../firebase';

function LoginForm({ setUser }) {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');

  const handleLogin = async () => {
    try {
      await firebase.auth().signInWithEmailAndPassword(email, password);
      setUser(firebase.auth().currentUser);
    } catch (error) {
      console.error('Error logging in:', error);
      alert('Invalid email or password');
    }
  };

  const handleRegister = async () => {
    try {
      await firebase.auth().createUserWithEmailAndPassword(email, password);
      setUser(firebase.auth().currentUser);
    } catch (error) {
      console.error('Error registering:', error);
      alert('Email already exists or invalid');
    }
  };

  return (
    <div>
      <input
        type="email"
        value={email}
        onChange={(e) => setEmail(e.target.value)}
        placeholder="Email"
      />
      <input
        type="password"
        value={password}
        onChange={(e) => setPassword(e.target.value)}
        placeholder="Password"
      />
      <button onClick={handleLogin}>Login</button>
      <button onClick={handleRegister}>Register</button>
    </div>
  );
}

export default LoginForm;
```

```javascript
function App() {

  );
```

## רגולריזציה בלמידת מכונה: סוגים, נוסחאות, קוד, יתרונות וחסרונות

רגולריזציה היא טכניקה חשובה בלמידת מכונה שמטרתה למנוע overfitting - מצב בו המודל לומד את נתוני האימון בצורה מושלמת מדי, אך מתקשה להכליל לנתונים חדשים. רגולריזציה מוסיפה עונש למודל על מורכבותו, ובכך מעודדת אותו ללמוד מודלים פשוטים יותר שמספקים ביצועים טובים יותר על נתונים חדשים. הנה פירוט של הסוגים השונים של רגולריזציה:

**1. L1 Regularization (Lasso Regularization)**

*   **הנוסחה:** `Cost = Loss + λ * Σ |wᵢ|`, כאשר:
    *   `Loss` - פונקציית המחיר של המודל (למשל, MSE עבור רגרסיה ליניארית).
    *   `λ` - פרמטר הרגולריזציה, השולט על עוצמת העונש. ערך גבוה יותר של `λ` מעודד פשטות יותר.
    *   `wᵢ` - משקלים של המודל.
    *   `Σ |wᵢ|` - סכום הערכים המוחלטים של המשקלים.
*   **ההשפעה:** L1 רגולריזציה מעודדת את המודל לקבוע חלק מהמשקלים ל-0. זה גורם ליצירת מודל דליל - מודל שבו מעט מאוד משקלים שונים מאפס.
*   **יתרונות:**
    *   דלילות: עוזרת לבחור את התכונות החשובות ביותר.
    *   הפחתת overfitting.
    *   יעילות חישובית: מודלים דלילים צורכים פחות משאבים.
*   **חסרונות:**
    *   קשה ליישום: ייתכן שצריך לבצע חיפוש מקיף על פני ערכים שונים של `λ`.
    *   לא תמיד מדויקת: עשויה לבחור תכונות לא רלוונטיות.
*   **דוגמת קוד (Python):**

```python
from sklearn.linear_model import Lasso
from sklearn.datasets import make_classification
import numpy as np

# צור נתוני סימולציה
X, y = make_classification(n_samples=100, n_features=10, random_state=42)

# הגדר מודל Lasso עם פרמטר רגולריזציה של 0.1
lasso = Lasso(alpha=0.1)

# אמן את המודל
lasso.fit(X, y)

# הדפס את המשקלים של המודל
print(lasso.coef_)
```

**2. L2 Regularization (Ridge Regression)**

*   **הנוסחה:** `Cost = Loss + λ * Σ wᵢ²`, כאשר:
    *   `Loss` - פונקציית המחיר של המודל.
    *   `λ` - פרמטר הרגולריזציה.
    *   `wᵢ` - משקלים של המודל.
    *   `Σ wᵢ²` - סכום ריבועי המשקלים.
*   **ההשפעה:** L2 רגולריזציה מעודדת את המודל לקטון את גודל המשקלים. המשקלים לא בהכרח יהפכו ל-0, אך הם יהיו קטנים יותר.
*   **יתרונות:**
    *   הפחתת overfitting.
    *   פשטות יישום: לרוב ניתן למצוא ערך מתאים ל-`λ` באמצעות ניסוי וטעייה.
    *   יציבות: פחות רגישה לשינויים קטנים בנתוני האימון.
*   **חסרונות:**
    *   לא עוזרת בבחירת תכונות: המשקלים מצטמצמים אך לא בהכרח יהפכו ל-0.
*   **דוגמת קוד (Python):**

```python
from sklearn.linear_model import Ridge
from sklearn.datasets import make_classification
import numpy as np

# צור נתוני סימולציה
X, y = make_classification(n_samples=100, n_features=10, random_state=42)

# הגדר מודל Ridge עם פרמטר רגולריזציה של 0.1
ridge = Ridge(alpha=0.1)

# אמן את המודל
ridge.fit(X, y)

# הדפס את המשקלים של המודל
print(ridge.coef_)
```

**3. Elastic Net Regularization**

*   **הנוסחה:** `Cost = Loss + λ₁ * Σ |wᵢ| + λ₂ * Σ wᵢ²`, כאשר:
    *   `Loss` - פונקציית המחיר של המודל.
    *   `λ₁` - פרמטר הרגולריזציה עבור L1 רגולריזציה.
    *   `λ₂` - פרמטר הרגולריזציה עבור L2 רגולריזציה.
    *   `Σ |wᵢ|` - סכום הערכים המוחלטים של המשקלים.
    *   `Σ wᵢ²` - סכום ריבועי המשקלים.
*   **ההשפעה:** Elastic Net היא שילוב של L1 ו-L2 רגולריזציה. היא מאפשרת למודל ליהנות מהיתרונות של שתי השיטות - דלילות ויציבות.
*   **יתרונות:**
    *   שילוב של היתרונות של L1 ו-L2.
    *   יעילה במיוחד כאשר יש קורלציה גבוהה בין התכונות.
*   **חסרונות:**
    *   מורכבת יותר ליישום: צריך למצוא ערכים מתאימים לשני פרמטרי הרגולריזציה (`λ₁` ו `λ₂`).
*   **דוגמת קוד (Python):**

```python
from sklearn.linear_model import ElasticNet
from sklearn.datasets import make_classification

# הגדר מודל Elastic Net עם פרמטר רגולריזציה של 0.1 עבור L1 ו-0.1 עבור L2
elastic_net = ElasticNet(alpha=0.1, l1_ratio=0.1)
```

**4. Dropout Regularization (בנירונים):**

*   **יתרונות:**
    *   מניעת overfitting בצורה יעילה.
    *   פשוטה ליישום.
*   **חסרונות:**
    *   דורשת חישובים רבים: ייתכן שצריך להגדיל את זמן האימון.

```python
import tensorflow as tf
```
