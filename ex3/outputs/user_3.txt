import React, { useState } from 'react';
import { db } from '../firebase';
import { collection, addDoc } from 'firebase/firestore';

function ChatInput({ user }) {
  const [message, setMessage] = useState('');

  const sendMessage = async () => {
    if (message.trim()) {
      try {
        await addDoc(collection(db, 'messages'), {
          text: message,
          user: user?.uid,
          timestamp: Date.now(),
        });
        setMessage('');
      } catch (error) {
        console.error('Error sending message:', error);
      }
    }
  };

  return (
    <div className="chat-input">
      <input
        type="text"
        placeholder="Type your message..."
        value={message}
        onChange={(e) => setMessage(e.target.value)}
      />
      <button onClick={sendMessage}>Send</button>
    </div>
  );
}

import React, { useEffect, useState } from 'react';
import { db } from '../firebase';
import { collection, orderBy, query, onSnapshot } from 'firebase/firestore';

function ChatMessages({ user }) {
  const [messages, setMessages] = useState([]);

  useEffect(() => {
    const q = query(collection(db, 'messages'), orderBy('timestamp'));
    const unsubscribe = onSnapshot(q, (snapshot) => {
      const newMessages = snapshot.docs.map((doc) => ({
        id: doc.id,
        ...doc.data(),
      }));
      setMessages(newMessages);
    });

    return () => unsubscribe(); // ניקוי ה-subscription כאשר הקומפוננטה מוסרת
  }, []);

  return (
    <div className="chat-messages">
      {messages.map((message) => (
        <div key={message.id} className={`message ${message.user === user?.uid ? 'self' : 'other'}`}>
          <p>{message.text}</p>
        </div>
      ))}
    </div>
  );
}

export default ChatMessages;

function App() {
  const [user, setUser] = useState(null);

  useEffect(() => {
    const unsubscribe = auth.onAuthStateChanged((currentUser) => {
      setUser(currentUser);
    });

    return () => unsubscribe(); // ניקוי ה-subscription בעת הניתוק של הקומפוננטה

  if (!user) {

  }

  return (

      <ChatMessages user={user} />
      <ChatInput user={user} />
    </div>
  );
}
