```javascript
import React, { useState, useEffect } from 'react';
import { auth, db } from './firebaseConfig';
import { collection, query, orderBy, addDoc, onSnapshot } from 'firebase/firestore';

function Chat() {
  const [messages, setMessages] = useState([]);
  const [inputText, setInputText] = useState('');

  useEffect(() => {
    const collectionRef = collection(db, 'messages');
    const q = query(collectionRef, orderBy('createdAt'));

    const unsubscribe = onSnapshot(q, (snapshot) => {
      const newMessageList = snapshot.docs.map((doc) => ({
        id: doc.id,
        ...doc.data(),
      }));
      setMessages(newMessageList);
    });

    return () => unsubscribe();
  }, []);

  const sendMessage = async () => {
    if (inputText.trim() !== '') {
      try {
        const collectionRef = collection(db, 'messages');
        await addDoc(collectionRef, {
          text: inputText,
          createdAt: new Date(),
          userId: auth.currentUser.uid,
        });
        setInputText('');
      } catch (error) {
        console.error('Error sending message:', error);
      }
    }
  };

  return (
    <div>
      <h1>Chat App</h1>
      <div style={{ height: '300px', overflowY: 'scroll', border: '1px solid #ccc', padding: '10px' }}>
        {messages.map((message) => (
          <div key={message.id}>
            {message.userId === auth.currentUser.uid ? (
              <div style={{ textAlign: 'right' }}><strong>You:</strong></div>
            ) : (
              <div style={{ textAlign: 'left' }}><strong>Other:</strong></div>
            )}
            <p>{message.text}</p>
          </div>
        ))}
      </div>
      <input
        type="text"
        value={inputText}
        onChange={(e) => setInputText(e.target.value)}
        placeholder="Type your message..."
      />
      <button onClick={sendMessage}>Send</button>
    </div>
  );
}

export default Chat;
```
