    import { collection, query, orderBy, onSnapshot } from 'firebase/firestore';

    function Chat() {
      const [messages, setMessages] = useState([]);

      useEffect(() => {
        const q = query(collection(db, 'messages'), orderBy('createdAt'));
        onSnapshot(q, (querySnapshot) => {

            type="text"
            value={text}
            onChange={(e) => setText(e.target.value)}

    import React, { createContext, useContext, useState, useEffect } from 'react';
    import { auth } from '../firebase';

    const AuthContext = createContext();

    export function useAuth() {
      return useContext(AuthContext);
    }

    export function AuthProvider({ children }) {
      const [user, setUser] = useState(null);

      useEffect(() => {
        const unsubscribe = auth.onAuthStateChanged((user) => {
          setUser(user);
        });

        return () => unsubscribe();
      }, []);

      return (

    }

## רגולריזציה בלמידת מכונה: מדריך מפורט

רגולריזציה (Regularization) היא טכניקה חשובה בלמידת מכונה המיועדת למנוע overfitting - מצב בו המודל לומד את נתוני האימון בצורה מושלמת מדי ואינו מצליח להכליל בצורה טובה לנתונים חדשים. הרגולריזציה מוסיפה עונש למודל על מורכבותו, ובכך מעודדת אותו ללמוד מודל פשוט יותר שיכול להכליל בצורה טובה יותר.

הנה סקירה מפורטת של סוגי הרגולריזציה השונים, עם נוסחאות, דוגמאות קוד, יתרונות וחסרונות:

**1. L1 Regularization (Lasso Regression):**

*   **עקרון:** מוסיפה את סכום הערכים המוחלטים של המשקלים למודל.
*   **נוסחה:** `loss = MSE + λ * Σ |w_i|`, כאשר `MSE` הוא ממוצע שגיאות ריבועיות, `λ` הוא פרמטר הרגולריזציה ו-`w_i` הם המשקלים של המודל.
*   **יתרונות:**
    *   יעילה בבחירת משתנים -  מכריחה חלק מהמשקלים להיות בדיוק אפס, ובכך מבצעת בחירת משתנים באופן אוטומטי.
    *   יעילה במקרים בהם יש הרבה משתנים לא רלוונטיים.
*   **חסרונות:**
    *   פחות מדויקת אם כל המשתנים רלוונטיים.
    *   חישוב יכול להיות יקר יותר מבחינה חישובית.
*   **דוגמת קוד (Python - scikit-learn):**
    ```python
    from sklearn.linear_model import Lasso
    from sklearn.datasets import make_regression

    # יצירת נתונים לדוגמה
    X, y = make_regression(n_samples=100, n_features=10, random_state=42)

    # יצירת מודל Lasso
    lasso = Lasso(alpha=0.1)  # α הוא פרמטר הרגולריזציה

    # אימון המודל
    lasso.fit(X, y)

    # הצגת המשקלים של המודל
    print(lasso.coef_)
    ```

**2. L2 Regularization (Ridge Regression):**

*   **עקרון:** מוסיפה את סכום ריבועי המשקלים למודל.
*   **נוסחה:** `loss = MSE + λ * Σ w_i^2`, כאשר `MSE` הוא ממוצע שגיאות ריבועיות, `λ` הוא פרמטר הרגולריזציה ו-`w_i` הם המשקלים של המודל.
*   **יתרונות:**
    *   יעילה במניעת overfitting.
    *   יותר מדויקת כאשר כל המשתנים רלוונטיים.
    *   יציבה מבחינה חישובית.
*   **חסרונות:**
    *   לא מבצעת בחירת משתנים - כל המשקלים קטנים, אך לא אפס.
*   **דוגמת קוד (Python - scikit-learn):**
    ```python
    from sklearn.linear_model import Ridge
    from sklearn.datasets import make_regression

    # יצירת נתונים לדוגמה
    X, y = make_regression(n_samples=100, n_features=10, random_state=42)

    # יצירת מודל Ridge
    ridge = Ridge(alpha=0.1)  # α הוא פרמטר הרגולריזציה

    # אימון המודל
    ridge.fit(X, y)

    # הצגת המשקלים של המודל
    print(ridge.coef_)
    ```

**3. Elastic Net Regularization:**

*   **עקרון:** שילוב של L1 ו-L2 רגולריזציה.
*   **נוסחה:** `loss = MSE + λ₁ * Σ |w_i| + λ₂ * Σ w_i^2`, כאשר `MSE` הוא ממוצע שגיאות ריבועיות, `λ₁` ו-`λ₂` הם פרמטרי הרגולריזציה ו-`w_i` הם המשקלים של המודל.
*   **יתרונות:**
    *   משלבת את היתרונות של L1 ו-L2 רגולריזציה.
    *   יכולה לבצע בחירת משתנים גם במקרים שבהם יש קורלציה גבוהה בין המשתנים.
*   **חסרונות:**
    *   מצריכה כוונון של שני פרמטרים (`λ₁` ו-`λ₂`).
*   **דוגמת קוד (Python - scikit-learn):**
    ```python
    from sklearn.linear_model import ElasticNet
    from sklearn.datasets import make_regression

    X, y = make_regression(n_samples=100, n_features=10, random_state=42)

    # יצירת מודל ElasticNet
    elastic_net = ElasticNet(alpha=0.1, l1_ratio=0.5)  # α הוא פרמטר הרגולריזציה, l1_ratio מציין את היחס בין L1 ל-L2

    # אימון המודל
    elastic_net.fit(X, y)

    # הצגת המשקלים של המודל
    print(elastic_net.coef_)
    ```

*   **חסרונות:**
    *   דורשת כוונון של פרמטר dropout (הסיכוי להסיר נוירון).
