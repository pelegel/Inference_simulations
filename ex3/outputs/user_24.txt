import { useFirebase } from 'react-firebase-hooks';

function ChatForm({ ref }) {
  const [message, setMessage] = useState('');

  const { firebase } = useFirebase();

  const handleSubmit = async (e) => {
    e.preventDefault();
    if (message.trim() !== '') {
      firebase.firestore().collection('chats').add({
        text: message,
        createdAt: firebase.firestore.FieldValue.serverTimestamp(),
        uid: firebase.auth().currentUser.uid,
      });
      setMessage('');
      ref.current?.scrollToEnd(); // Scroll down after submitting
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        type="text"
        value={message}
        onChange={(e) => setMessage(e.target.value)}
        placeholder="כתוב הודעה..."
      />
      <button type="submit">שלח</button>
    </form>
  );
}

export default ChatForm;

function ChatWindow() {
  const { firebase } = useFirebase();
  const chatRef = useRef(null);
  const [messages, setMessages] = useState([]);

  useEffect(() => {
    const unsubscribe = firebase
      .firestore()
      .collection('chats')
      .orderBy('createdAt')
      .onSnapshot((snapshot) => {
        const newMessages = snapshot.docs.map((doc) => ({
          id: doc.id,
          ...doc.data(),
        }));
        setMessages(newMessages);
      });

    return () => unsubscribe(); // Unsubscribe when component unmounts
  }, [firebase]);

  return (
    <div className="chat-window">
      <div className="messages" ref={chatRef}>
        {messages.map((message) => (

## רגולריזציה בלמידת מכונה: מדריך מפורט

רגולריזציה היא טכניקה חשובה בלמידת מכונה שמטרתה למנוע overfitting - מצב שבו המודל לומד את הנתונים בצורה מוגזמת ומאבד את היכולת להכליל לנתונים חדשים. רגולריזציה מוסיפה עונש למודל על מורכבותו, ובכך מעודדת אותו ללמוד מודלים פשוטים יותר שיכולים להכליל בצורה טובה יותר.

הנה סקירה מפורטת של סוגי הרגולריזציה הנפוצים ביותר:

**1. L1 Regularization (LASSO):**

*   **העיקרון:** מוסיף עונש פרופורציונלי לסכום הערכים המוחלטים של המשקלים במודל.
*   **נוסחה:**  `Loss = Loss_function + λ * Σ |w_i|`, כאשר `λ` הוא פרמטר הרגולריזציה ו-`w_i` הם המשקלים במודל.
*   **יתרונות:**
    *   מעודדת sparsity - חלק מהמשקלים במודל יהפכו לאפס, מה שמקטין את מורכבות המודל ומסייע בבחירת תכונות (feature selection).
    *   יעילה במיוחד במקרים שבהם יש מספר רב של תכונות שרבות מהן לא רלוונטיות.
*   **חסרונות:**
    *   יכולה להיות רגישה לערכים חריגים בנתונים.
    *   הבחירה של הערך האופטימלי עבור `λ` דורשת ניסוי וטעייה.
*   **דוגמת קוד (Python):**

```python
from sklearn.linear_model import Lasso
from sklearn.datasets import make_classification

# יצירת נתונים סימולטיביים
X, y = make_classification(n_samples=100, n_features=10, random_state=42)

# יצירת מודל Lasso
lasso = Lasso(alpha=0.1)  # alpha הוא פרמטר הרגולריזציה λ

# אימון המודל
lasso.fit(X, y)

# הצגת המשקלים
print(lasso.coef_)
```

**2. L2 Regularization (Ridge):**

*   **העיקרון:** מוסיף עונש פרופורציונלי לסכום הריבועים של המשקלים במודל.
*   **נוסחה:**  `Loss = Loss_function + λ * Σ w_i^2`, כאשר `λ` הוא פרמטר הרגולריזציה ו-`w_i` הם המשקלים במודל.
*   **יתרונות:**
    *   יציבה יותר מ-L1 regularization.
    *   מונעת מהמשקלים להתאפס לחלוטין, מה שמבטיח שהמודל לא יאבד מידע חשוב.
    *   יעילה במקרים שבהם יש קורלציה גבוהה בין התכונות.
*   **חסרונות:**
    *   לא מעודדת sparsity - כל המשקלים במודל יישארו קטנים אך שונים מאפס.
    *   פחות יעילה בבחירת תכונות.
*   **דוגמת קוד (Python):**

```python
from sklearn.linear_model import Ridge
from sklearn.datasets import make_classification

# יצירת נתונים סימולטיביים

# אימון המודל
ridge.fit(X, y)

# הצגת המשקלים
print(ridge.coef_)
```

**3. Elastic Net:**

*   **העיקרון:** שילוב של L1 ו-L2 regularization. מוסיף עונש פרופורציונלי לסכום הערכים המוחלטים וריבועי המשקלים במודל.
*   **נוסחה:**  `Loss = Loss_function + λ_1 * Σ |w_i| + λ_2 * Σ w_i^2`, כאשר `λ_1` ו-`λ_2` הם פרמטרי הרגולריזציה.
*   **יתרונות:**
    *   משלב את היתרונות של L1 ו-L2 regularization.
    *   יעיל במקרים שבהם יש קורלציה גבוהה בין התכונות ורבים מהן לא רלוונטיות.
    *   מספק גמישות רבה יותר בבחירת הפרמטרים.
*   **חסרונות:**
    *   דורש בחירה של שני פרמטרים ( `λ_1` ו- `λ_2`), מה שמסבך את תהליך הכיוונון.
*   **דוגמת קוד (Python):**

```python
from sklearn.linear_model import ElasticNet
from sklearn.datasets import make_classification

# יצירת נתונים סימולטיביים
X, y = make_classification(n_samples=100, n_features=10, random_state=42)

# יצירת מודל Elastic Net
elastic_net = ElasticNet(alpha=0.1, l1_ratio=0.5)  # alpha הוא פרמטר הרגולריזציה λ, l1_ratio הוא היחס בין L1 ל-L2

# אימון המודל
elastic_net.fit(X, y)

# הצגת המשקלים
print(elastic_net.coef_)
```
