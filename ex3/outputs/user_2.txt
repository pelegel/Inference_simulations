import { signInWithPopup, GoogleAuthProvider } from 'firebase/auth';
import { auth } from '../firebase';

function LoginForm({ setUser }) {
  const googleProvider = new GoogleAuthProvider();

  const signInWithGoogle = async () => {
    try {
      const result = await signInWithPopup(auth, googleProvider);
      setUser({
        uid: result.user.uid,
        displayName: result.user.displayName,
        email: result.user.email,

export default LoginForm;

// src/components/ChatForm.js
import React, { useState } from 'react';
import { addDoc, collection } from 'firebase/firestore';
import { db } from '../firebase';

function ChatForm({ user }) {
  const [message, setMessage] = useState('');

  const sendMessage = async () => {
    if (message) {
      try {
        await addDoc(collection(db, 'chats'), {
          text: message,
          uid: user.uid,
          displayName: user.displayName,
          createdAt: new Date().getTime(),
        });
        setMessage('');
      } catch (error) {
        console.error('Error adding message:', error);
      }
    }
  };

  return (
    <div>
      <input
        type="text"
        value={message}
        onChange={(e) => setMessage(e.target.value)}
      />
      <button onClick={sendMessage}>Send</button>
    </div>
  );
}

export default ChatForm;

// src/components/ChatMessages.js
import React, { useEffect, useState } from 'react';
import { collection, query, orderBy, onSnapshot } from 'firebase/firestore';
import { db } from '../firebase';

function ChatMessages() {
  const [messages, setMessages] = useState([]);

  useEffect(() => {
    const q = query(collection(db, 'chats'), orderBy('createdAt'));
    const unsubscribe = onSnapshot(q, (querySnapshot) => {
      const newMessages = querySnapshot.docs.map((doc) => ({ ...doc.data(), id: doc.id }));
      setMessages(newMessages);
    });

    return () => unsubscribe();
  }, []);

  return (
    <div>
      {messages.map((message) => (
        <div key={message.id}>
          <strong>{message.displayName}:</strong> {message.text}
        </div>
      ))}
    </div>
  );
}

export default ChatMessages;

// src/components/App.js
import React, { useState } from 'react';
import LoginForm from './LoginForm';
import ChatForm from './ChatForm';
import ChatMessages from './ChatMessages';

function App() {

  return (

        </div>

  );
}

## Regularization בלמידת מכונה: מדריך מפורט

Regularization הוא טכניקה חשובה בלמידת מכונה שמטרתה למנוע overfitting (התאמת יתר) של המודל לנתוני האימון. overfitting מתרחש כאשר המודל לומד את הרעשים בנתונים במקום את הקשרים האמיתיים. regularization מוסיפה עונש למודלים מורכבים, ובכך מעודדת את המודל לבחור את הפתרון הפשוט ביותר שמספק ביצועים טובים על נתוני האימון וגם על נתוני הבדיקה. הנה פירוט מפורט של סוגי ה-regularization השונים, עם נוסחאות, דוגמאות קוד, יתרונות וחסרונות:

**1. L1 Regularization (Lasso Regression):**

*   **עיקרון:** מוסיף את הסכום המוחלט של המקדמים של המודל לפונקציית העלות.
*   **נוסחה:**
    *   Loss = MSE + λ * Σ |wᵢ|
    *   MSE = Mean Squared Error
    *   λ = פרמטר ה-regularization. קובע את עוצמת העונש על המקדמים.
    *   wᵢ = המקדם של המאפיין ה-i.
*   **יתרונות:**
    *   **Feature Selection:** L1 regularization יכול לגרום למקדמים של מאפיינים לא רלוונטיים להיות אפס. זה הופך אותו לטכניקה יעילה לבחירת מאפיינים (feature selection).
    *   **מודלים דלילים:** יוצר מודלים דלילים (sparse) עם מעט מקדמים לא אפסים.
*   **חסרונות:**
    *   **לא תמיד יציב:** עלול להיות לא יציב מבחינה מספרית.
    *   **קשה יותר ליישם:** יחסית ל-L2 regularization.
*   **דוגמת קוד (Python עם scikit-learn):**

```python
from sklearn.linear_model import Lasso
from sklearn.datasets import make_regression
import numpy as np

# יצירת נתונים לדוגמה
X, y = make_regression(n_samples=100, n_features=10, noise=0.1)

# הגדרת מודל L1 regularization
lasso = Lasso(alpha=1.0) # alpha הוא פרמטר ה-regularization

# אימון המודל
lasso.fit(X, y)

# הדפסת המקדמים
print(lasso.coef_)
```

**2. L2 Regularization (Ridge Regression):**

*   **עיקרון:** מוסיף את הסכום הריבועי של המקדמים של המודל לפונקציית העלות.
*   **נוסחה:**
    *   Loss = MSE + λ * Σ wᵢ²
    *   MSE = Mean Squared Error
    *   λ = פרמטר ה-regularization. קובע את עוצמת העונש על המקדמים.
    *   wᵢ = המקדם של המאפיין ה-i.
*   **יתרונות:**
    *   **יציבות:** יציבה מבחינה מספרית.
    *   **יעילות:** יחסית קלה ליישום.
    *   **מניעת overfitting:** יעילה במניעת overfitting.
*   **חסרונות:**
    *   **לא מבצעת feature selection:** לא גורמת למקדמים של מאפיינים לא רלוונטיים להיות אפס.
    *   **מודלים צפופים:** יוצרת מודלים צפופים (dense) עם כל המקדמים שונים מאפס.
*   **דוגמת קוד (Python עם scikit-learn):**

```python
from sklearn.linear_model import Ridge
from sklearn.datasets import make_regression
import numpy as np

# יצירת נתונים לדוגמה
X, y = make_regression(n_samples=100, n_features=10, noise=0.1)

# הגדרת מודל L2 regularization
ridge = Ridge(alpha=1.0) # alpha הוא פרמטר ה-regularization

# אימון המודל
ridge.fit(X, y)

# הדפסת המקדמים
print(ridge.coef_)
```

**3. Elastic Net Regularization:**

*   **עיקרון:** שילוב של L1 ו-L2 regularization. מוסיף גם את הסכום המוחלט וגם את הסכום הריבועי של המקדמים לפונקציית העלות.
*   **נוסחה:**
    *   Loss = MSE + λ₁ * Σ |wᵢ| + λ₂ * Σ wᵢ²
    *   MSE = Mean Squared Error
    *   λ₁ = פרמטר ה-regularization עבור L1.
    *   λ₂ = פרמטר ה-regularization עבור L2.
    *   wᵢ = המקדם של המאפיין ה-i.
*   **יתרונות:**
    *   **הטבות של שתי הטכניקות:** משלב את היתרונות של L1 ו-L2 regularization.
    *   **בחירת מאפיינים:** יכול לבצע בחירת מאפיינים כמו L1 regularization.
    *   **יציבות:** יציבה יותר מ-L1 regularization.
*   **חסרונות:**
    *   **מורכבות:** יותר מורכב ליישם מ-L1 ו-L2 regularization.
    *   **צריך לכייל שני פרמטרים:** צריך לקבוע את הערכים המתאימים של λ₁ ו-λ₂.
*   **דוגמת קוד (Python עם scikit-learn):**

```python
from sklearn.linear_model import ElasticNet
from sklearn.datasets import make_regression
import numpy as np

# יצירת נתונים לדוגמה
X, y = make_regression(n_samples=100, n_features=10, noise=0.1)

# הגדרת מודל Elastic Net regularization
elastic_net = ElasticNet(alpha=1.0, l1_ratio=0.5) # alpha הוא פרמטר ה-regularization, l1_ratio שולט על היחס בין L1 ו-L2

# אימון המודל
elastic_net.fit(X, y)

# הדפסת המקדמים
print(elastic_net.coef_)
```

**4. Dropout:**

*   **עיקרון:** טכניקה שמשמשת בעיקר ברשתות נוירונים. במהלך האימון, היא משביתה באופן אקראי חלק מהנוירונים (יחידות חישוב). זה מונע מהנוירונים להיות תלויים אחד בשני ומעודד למידה של תכונות עצמאיות.
*   **יתרונות:**
    *   **מניעת overfitting:** יעילה במניעת overfitting.
    *   **יצירת מודלים עמידים:** עוזרת ליצור מודלים עמידים יותר להפרעות ורעשים.
*   **חסרונות:**
    *   **עלולה להאריך את זמן האימון:** מכיוון שהיא דורשת אימון של מספר מודלים שונים.
    *   **דורשת כיוונון:** צריך לכייל את קצב ה-dropout.
*   **דוגמת קוד (Python עם TensorFlow/Keras):**

```python
import tensorflow as tf

# הגדרת מודל רשת נוירונים
model = tf.keras.Sequential([
    tf.keras.layers.Dense(128, activation='relu', input_shape=(10,)),
    tf.keras.layers.Dropout(0.2), # הגדרת שכבת dropout עם קצב של 0.2
    tf.keras.layers.Dense(1)
])

# קומפילציה של המודל
model.compile(optimizer='adam', loss='mse')
```

**בחירת ה-Regularization המתאים:**

הבחירה בין סוגי ה-regularization השונים תלויה בנתונים ובמודל. הנה כמה הנחיות כלליות:
