**שלב 5: בניית רכיבי React**

צור תיקייה בשם `components` ושם הקובץ `ChatRoom.js`:

```javascript
// components/ChatRoom.js
import React, { useState, useEffect } from 'react';
import { useCollection } from 'react-firebase-hooks';
import { db } from '../services/firebaseService';
import { collection, orderBy } from 'firebase/firestore';

function ChatRoom() {
  const [messages, loading, error] = useCollection(
    collection(db, 'messages'),
    {
      orderBy: 'createdAt'
    }
  );

  const [newMessage, setNewMessage] = useState('');

  const sendMessage = async () => {
    if (newMessage) {
      try {
        await db.collection('messages').add({
          text: newMessage,
          createdAt: new Date(),
          userId: 'user_id' // Replace with actual user ID if authentication is implemented
        });
        setNewMessage('');
      } catch (error) {
        console.error('Error sending message:', error);
      }
    }
  };

  return (
    <div>
      <h1>Chat Room</h1>
      <div style={{ height: '300px', overflowY: 'scroll' }}>
        {loading ? (
          <p>Loading...</p>
        ) : error ? (
          <p>Error: {error.message}</p>
        ) : messages?.map((message) => (
          <div key={message.id}>
            {message.text} - {new Date(message.createdAt).toLocaleTimeString()}
          </div>
        ))}
      </div>
      <input
        type="text"
        value={newMessage}
        onChange={(e) => setNewMessage(e.target.value)}
      />
      <button onClick={sendMessage}>Send</button>
    </div>
  );
}

export default ChatRoom;
```
