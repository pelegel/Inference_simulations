```javascript
// src/components/Chat.js
import React, { useState, useEffect } from 'react';
import { getFirestore } from 'firebase/firestore';
import Message from './Message';

const db = getFirestore();

function Chat() {
  const [messages, setMessages] = useState([]);
  const [text, setText] = useState('');

  useEffect(() => {
    const collectionRef = collection(db, 'messages');
    const unsubscribe = onSnapshot(collectionRef, (snapshot) => {
      const newMessageList = snapshot.docs.map((doc) => ({
        id: doc.id,
        text: doc.data().text,
      }));
      setMessages(newMessageList);
    });

    return () => unsubscribe();
  }, []);

  const sendMessage = async (event) => {
    event.preventDefault();

    const collectionRef = collection(db, 'messages');
    await addDoc(collectionRef, { text: text });

    setText('');
  };

  return (
    <div>
      <form onSubmit={sendMessage}>
        <input
          type="text"
          value={text}
          onChange={(e) => setText(e.target.value)}
          placeholder="Write your message..."
        />
        <button type="submit">Send</button>
      </form>

      <ul>
        {messages.map((message) => (
          <Message key={message.id} message={message} />
        ))}
      </ul>
    </div>
  );
}

export default Chat;

// src/components/Message.js
function Message({ message }) {
  return <li>{message.text}</li>;
}

export default Message;
```

## רגולריזציה בלמידת מכונה: מדריך מפורט

רגולריזציה היא טכניקה חשובה בלמידת מכונה שמטרתה למנוע overfitting - מצב שבו המודל לומד את נתוני האימון בצורה מושלמת מדי ואינו מצליח להכליל טוב על נתונים חדשים. רגולריזציה מוסיפה עונש למודל על מורכבותו, ובכך מעודדת אותו ללמוד פתרונות פשוטים יותר. הנה סקירה של סוגי הרגולריזציה העיקריים:

**1. L1 Regularization (Lasso Regularization)**

*   **העיקרון:** מוסיפה את סכום הערכים המוחלטים של משקלי המודל לפונקציית העלות.
*   **נוסחה:**  `Cost = MSE + λ * Σ |wᵢ|` (MSE - Mean Squared Error, λ - פרמטר הרגולריזציה, wᵢ - משקל המודל).
*   **יתרונות:**
    *   **ייצור sparsity:** L1 רגולריזציה נוטה לדחוף משקלים רבים לאפס, מה שמייצר מודל דליל (sparse). זה יכול להיות שימושי בבחירת תכונות (feature selection).
    *   **פשוטה ליישום:** יחסית קלה להבנה וליישום.
*   **חסרונות:**
    *   **לא תמיד יציבה:** עלולה להיות לא יציבה בבעיות עם קולינאריות גבוהה.
    *   **קשה לפתור:**  הנוסחה אינה ניתנת לגזירה בקלות בכל נקודה, מה שמקשה על אלגוריתמים מסוימים למצוא את הפתרון האופטימלי.
*   **דוגמה קוד (Python - Scikit-learn):**

```python
from sklearn.linear_model import Lasso
import numpy as np

# דוגמה של נתונים
X = np.random.rand(100, 5)
y = np.random.rand(100)

# יצירת מודל Lasso
lasso = Lasso(alpha=0.1)  # α הוא פרמטר הרגולריזציה

# אימון המודל
lasso.fit(X, y)

# הצגת המשקלים של המודל
print(lasso.coef_)
```

**2. L2 Regularization (Ridge Regression)**

*   **העיקרון:** מוסיפה את סכום ריבועי המשקלים של המודל לפונקציית העלות.
*   **נוסחה:** `Cost = MSE + λ * Σ wᵢ²` (MSE - Mean Squared Error, λ - פרמטר הרגולריזציה, wᵢ - משקל המודל).
*   **יתרונות:**
    *   **יציבה:** L2 רגולריזציה יציבה יותר מ-L1 ופחות רגישה לקולינאריות.
    *   **יעילה:** לרוב מובילה לביצועים טובים יותר מ-L1.
*   **חסרונות:**
    *   **לא מייצרת sparsity:** לא גורמת לדחיפה של משקלים לאפס.
    *   **יכולה להיות קשה לפרשנות:** מכיוון שכל המשקלים נשמרים (אף על פי שקטנים), המודל יכול להיות קשה יותר לפרשנות.
*   **דוגמה קוד (Python - Scikit-learn):**

```python
from sklearn.linear_model import Ridge
import numpy as np

# דוגמה של נתונים
X = np.random.rand(100, 5)
y = np.random.rand(100)

# יצירת מודל Ridge
ridge = Ridge(alpha=0.1)  # α הוא פרמטר הרגולריזציה

# אימון המודל
ridge.fit(X, y)

# הצגת המשקלים של המודל
print(ridge.coef_)
```

**3. Elastic Net Regularization**

*   **העיקרון:** שילוב של L1 ו-L2 רגולריזציה.  מוסיפה סכום של הערכים המוחלטים וריבועי המשקלים לפונקציית העלות.
*   **נוסחה:** `Cost = MSE + λ₁ * Σ |wᵢ| + λ₂ * Σ wᵢ²` (MSE - Mean Squared Error, λ₁ - פרמטר רגולריזציה L1, λ₂ - פרמטר רגולריזציה L2, wᵢ - משקל המודל).
*   **יתרונות:**
    *   **השגת הטוב מכל העולמות:** משלבת את היתרונות של L1 ו-L2.  ייצור sparsity (כמו L1) ויציבות (כמו L2).
    *   **גמישות:** מאפשרת שליטה על האיזון בין רגולריזציה L1 ו-L2.
*   **חסרונות:**
    *   **מורכבת יותר:** יש צורך להגדיר שני פרמטרים (λ₁ ו-λ₂).
    *   **יותר קשה ליישום:** דורשת אלגוריתמים מורכבים יותר ליישום.
*   **דוגמה קוד (Python - Scikit-learn):**

```python
from sklearn.linear_model import ElasticNet
import numpy as np

# דוגמה של נתונים
X = np.random.rand(100, 5)
y = np.random.rand(100)

# יצירת מודל ElasticNet
elasticnet = ElasticNet(alpha=0.1, l1_ratio=0.5)  # α - פרמטר הרגולריזציה, l1_ratio - היחס בין L1 ל-L2

# אימון המודל
elasticnet.fit(X, y)

# הצגת המשקלים של המודל
print(elasticnet.coef_)
```

**4. Dropout Regularization (במודלים של רשתות נוירונים)**

*   **העיקרון:** במהלך האימון, משביתים באופן אקראי נוירונים (עם הסתברות מסוימת). זה מונע ממודל הרשת הנוירונית להיות תלוי יתר על המידה בנוירונים ספציפיים.
*   **יתרונות:**
    *   **מניעת overfitting:** עוזרת למנוע overfitting על ידי הגברת הגיוון של המודל.
    *   **פשוטה ליישום:** יחסית קלה להוסיף למודל רשת נוירונים.
*   **חסרונות:**
    *   **יכולה להאט את האימון:** עקב הדרכה של מספר מודלים שונים (עם נוירונים שונים משובטים).
    *   **דורשת כיול:** יש צורך להגדיר את הסתברות ה-dropout בצורה נכונה.
*   **דוגמה קוד (Python - TensorFlow/Keras):**

```python
import tensorflow as tf
from tensorflow.keras.layers import Dense, Dropout
from tensorflow.keras.models import Sequential

    Dense(128, activation='relu', input_shape=(10,)),

    Dense(64, activation='relu'),
    Dropout(0.2),  # השבתה של 20% מהנוירונים באופן אקראי
    Dense(1)
```
