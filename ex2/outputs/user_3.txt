```javascript
import React, { useState, useEffect } from 'react';
import { getFirestore, collection, addDoc, onSnapshot, doc, getDoc } from 'firebase/firestore';
import { getAuth } from 'firebase/auth';

function Chat() {
  const [messages, setMessages] = useState([]);
  const [newMessage, setNewMessage] = useState('');
  const db = getFirestore();
  const auth = getAuth();
  const [user, setUser] = useState(null);

  useEffect(() => {
    const unsubscribe = onSnapshot(collection(db, 'messages'), (snapshot) => {
      const newMessages = snapshot.docs.map((doc) => ({
        id: doc.id,
        ...doc.data(),
      }));
      setMessages(newMessages);
    });

    return () => unsubscribe();
  }, [db]);

  useEffect(() => {
    const getProfile = async () => {
      const userRef = doc(db, "users", auth.currentUser?.uid);
      const docSnap = await getDoc(userRef);

      if (docSnap.exists()) {
        setUser(docSnap.data());
      } else {
        console.log("No such document!");
      }
    }

    if (auth.currentUser){
      getProfile();
    }
  }, [auth.currentUser]);

  const sendMessage = async (e) => {
    e.preventDefault();
    if (newMessage.trim() === '') return;

    const messagesRef = collection(db, 'messages');
    const msg = {
      sender: user?.displayName || auth.currentUser?.email,
      text: newMessage,
      timestamp: Date.now(),
    };

    await addDoc(messagesRef, msg);
    setNewMessage('');
  };

  return (
    <div>
      <h2>Chat</h2>
      <ul>
        {messages.map((message) => (
          <li key={message.id}>
            <strong>{message.sender}:</strong> {message.text} - {new Date(message.timestamp).toLocaleTimeString()}
          </li>
        ))}
      </ul>
      <form onSubmit={sendMessage}>
        <input
          type="text"
          value={newMessage}
          onChange={(e) => setNewMessage(e.target.value)}
        />
        <button type="submit">Send</button>
      </form>
    </div>
  );
}

export default Chat;
```

## Regularization בלמידת מכונה: מדריך מפורט

Regularization היא טכניקה חיונית בלמידת מכונה המשמשת כדי למנוע overfitting – מצב שבו מודל לומד את נתוני האימון בצורה מוגזמת, ומתקשה להכליל לנתונים חדשים. היא עושה זאת על ידי הוספת "עונש" לפונקציית העלות, שמכריח את המודל להיות פשוט יותר.

להלן סקירה מפורטת של סוגי regularization שונים:

**1. L1 Regularization (Lasso Regression):**

*   **עיקרון:** מוסיפה לפונקציית העלות איבר פרופורציונלי לערך המוחלט של המקדמים (weights) של המודל.
*   **נוסחה:**  Cost = Loss + λ * Σ|βᵢ|  (כאשר βᵢ הם המקדמים ו-λ הוא פרמטר regularization).
*   **דוגמת קוד (Python, scikit-learn):**
   ```python
   from sklearn.linear_model import Lasso
   from sklearn.model_selection import train_test_split
   from sklearn.datasets import make_regression

   # יצירת נתונים לדוגמה
   X, y = make_regression(n_samples=100, n_features=10, noise=0.1)
   X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)

   # יצירת מודל Lasso עם פרמטר regularization
   lasso = Lasso(alpha=0.1) # alpha = λ

   # אימון המודל
   lasso.fit(X_train, y_train)

   # חיזוי
   y_pred = lasso.predict(X_test)
   ```
*   **יתרונות:**
    *   **Feature Selection:**  גורם למקדמים של חלק מהתכונות (features) להיות אפס, ובכך מבצע בחירת תכונות באופן אוטומטי (sparse model).
    *   **מונע overfitting:** מפשט את המודל, משפר את היכולת להכליל.
*   **חסרונות:**
    *   יכול להסיר תכונות חשובות בצורה לא מושכלת (כאשר α גבוה מדי).

**2. L2 Regularization (Ridge Regression):**

*   **עיקרון:** מוסיפה לפונקציית העלות איבר פרופורציונלי לריבוע המקדמים של המודל.
*   **נוסחה:** Cost = Loss + λ * Σβᵢ²  (כאשר βᵢ הם המקדמים ו-λ הוא פרמטר regularization).
*   **דוגמת קוד (Python, scikit-learn):**
   ```python
   from sklearn.linear_model import Ridge
   from sklearn.model_selection import train_test_split
   from sklearn.datasets import make_regression

   # יצירת נתונים לדוגמה
   X, y = make_regression(n_samples=100, n_features=10, noise=0.1)
   X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)

   # יצירת מודל Ridge עם פרמטר regularization
   ridge = Ridge(alpha=0.1) # alpha = λ

   # אימון המודל
   ridge.fit(X_train, y_train)

   # חיזוי
   y_pred = ridge.predict(X_test)
   ```
*   **יתרונות:**
    *   **מונע overfitting:** מפשט את המודל, משפר את היכולת להכליל.
    *   **יציבות:** מפחית את הרגישות לשינויים קטנים בנתונים.
*   **חסרונות:**
    *   לא מבצע בחירת תכונות באופן ישיר (המקדמים לא יהפכו לאפס, אלא יקטנו).

**3. Elastic Net Regularization:**

*   **עיקרון:** שילוב של L1 ו-L2 regularization.
*   **נוסחה:** Cost = Loss + λ₁ * Σ|βᵢ| + λ₂ * Σβᵢ²  (כאשר λ₁ ו-λ₂ הם פרמטרי regularization עבור L1 ו-L2, בהתאמה).
*   **דוגמת קוד (Python, scikit-learn):**
   ```python
   from sklearn.linear_model import ElasticNet
   from sklearn.model_selection import train_test_split
   from sklearn.datasets import make_regression

   # יצירת נתונים לדוגמה
   X, y = make_regression(n_samples=100, n_features=10, noise=0.1)
   X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2)

   # יצירת מודל ElasticNet עם פרמטרי regularization
   elastic_net = ElasticNet(alpha=0.1, l1_ratio=0.5) # alpha = λ, l1_ratio: חלוקה בין L1 ו-L2

   # אימון המודל
   elastic_net.fit(X_train, y_train)

   # חיזוי
   y_pred = elastic_net.predict(X_test)
   ```
*   **יתרונות:**
    *   **משלב את היתרונות של L1 ו-L2:** מבצע בחירת תכונות (כמו L1) וגם מפחית overfitting (כמו L2).
    *   **יציבות:** פחות רגיש לבחירת הפרמטרים מאשר L1 או L2 בנפרד.
*   **חסרונות:**
    *   דורש כוונון שני פרמטרים regularization (λ₁ ו-λ₂).

**4. Dropout Regularization (משמש בעיקר ברשתות נוירונים):**

*   **עיקרון:** במהלך האימון, באופן אקראי מורידים חלק מהנוירונים (עם הסתברות מסוימת, למשל 0.5) בכל איטרציה. הדבר גורם לרשת ללמוד תכונות מגוונות יותר ולא להיות תלויה בנוירונים ספציפיים.
*   **דוגמת קוד (Python, TensorFlow/Keras):**
   ```python

   from tensorflow.keras.layers import Dense, Dropout

   model = Sequential([
       Dense(128, activation='relu', input_shape=(784,)),
       Dropout(0.2), # הסתברות של 20% להוריד נוירונים
       Dense(10, activation='softmax')
   ])
   ```
*   **יתרונות:**
    *   **מונע overfitting:** מפשט את המודל ומגביר את היכולת להכליל.
    *   **קל ליישום:** דורש רק הוספת שכבת Dropout לרשת הנוירונים.
*   **חסרונות:**
    *   עלול להאריך את זמן האימון.

**5. Early Stopping:**

*   **עיקרון:** עוצר את האימון כאשר ביצועי המודל על סט הvalidation מתחילים להידרדר, גם אם על סט האימון הוא ממשיך להשתפר.
*   **דוגמת קוד (Python, scikit-learn):**

   ```python
   X, y = make_classification(n_samples=100, n_features=10, random_state=42)
   ```
