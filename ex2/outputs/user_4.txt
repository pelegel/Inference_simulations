import React, { useState, useEffect, useCallback } from 'react';
import firebase from 'firebase/app';
import 'firebase/firestore';
import 'firebase/auth';
import Message from './components/Message';
import MessageForm from './components/MessageForm';
import { useAuthState } from 'react-firebase-hooks/auth';

// Initialize Firebase
const firebaseApp = firebase.initializeApp({
  apiKey: process.env.REACT_APP_apiKey,
  authDomain: process.env.REACT_APP_authDomain,
  projectId: process.env.REACT_APP_projectId,
  storageBucket: process.env.REACT_APP_storageBucket,
  messagingSenderId: process.env.REACT_APP_messagingSenderId,
  appId: process.env.REACT_APP_appId,
  measurementId: process.env.REACT_APP_measurementId,
});


const auth = firebaseApp.auth();

function App() {
  const [user] = useAuthState(auth);
  const [messages, setMessages] = useState([]);

  const sendMessage = useCallback(
    (message) => {
      const db = firebaseApp.firestore();
      db.collection('messages').add({
        author: user.email,
        message,
        timestamp: firebase.firestore.FieldValue.serverTimestamp(),
      });
    },
    [user]
  );

  useEffect(() => {
    const db = firebaseApp.firestore();
    const unsubscribe = db.collection('messages').orderBy('timestamp', 'asc').onSnapshot((snapshot) => {
      const newMessages = snapshot.docs.map((doc) => ({
        id: doc.id,
        ...doc.data(),
      }));
      setMessages(newMessages);
    });

    return () => unsubscribe();
  }, []);

  return (
    <div className="App">
      {user ? (
        <>
          <h1>צ'אט בזמן אמת</h1>
          <div className="messages">
            {messages.map((message) => (
              <Message key={message.id} message={message.message} author={message.author} />
            ))}
          </div>
          <MessageForm sendMessage={sendMessage} />
        </>
      ) : (
        <h2>אנא התחבר כדי להשתתף בצ'אט</h2>
      )}
    </div>
  );
}

export default App;

## רגולריזציה (Regularization) בלמידת מכונה: מדריך מפורט

רגולריזציה היא טכניקה חשובה בלמידת מכונה שמטרתה למנוע התאמת יתר (Overfitting) של מודלים, כלומר מצב שבו המודל לומד את רעשי הנתונים בנוסף ליחסי הגומלין האמיתיים בין המשתנים. היא עושה זאת על ידי הוספת "קנס" לפונקציית העלות (Cost Function) שאנו מנסים למזער במהלך האימון. קנס זה מעודד מודלים פשוטים יותר, אשר נוטים להיות פחות רגישים לרעשי הנתונים.

**1. סוגי רגולריזציה:**

*   **L1 רגולריזציה (Lasso Regression):**
    *   **נוסחה:**  הוספת איבר לפונקציית העלות בצורה של  `λ * Σ |βj|`, כאשר:
        *   `βj` הם המקדמים של המודל.
        *   `λ` הוא מקדם הרגולריזציה (hyperparameter) שקובע את עוצמת הקנס.
    *   **השפעה:**  גורם למקדמים מסוימים להיות בדיוק 0, ובכך מבצע בחירה של תכונות (feature selection).
    *   **יתרונות:**  פשטות, בחירת תכונות אוטומטית, מתאים לבעיות עם מספר רב של תכונות.
    *   **חסרונות:**  עלול להסיר תכונות חשובות אם `λ` גבוה מדי.
    *   **קוד (Python with Scikit-learn):**

```python
from sklearn.linear_model import Lasso
import numpy as np

# יצירת נתונים לדוגמה
X = np.random.rand(100, 10)
y = np.random.rand(100)

# אימון מודל Lasso עם מקדם רגולריזציה של 0.1
model = Lasso(alpha=0.1)
model.fit(X, y)

# הדפסת המקדמים
print(model.coef_)
```

*   **L2 רגולריזציה (Ridge Regression):**
    *   **נוסחה:**  הוספת איבר לפונקציית העלות בצורה של `λ * Σ βj²`, כאשר:
        *   `βj` הם המקדמים של המודל.
        *   `λ` הוא מקדם הרגולריזציה.
    *   **השפעה:**  מצמצם את גודל המקדמים, אבל לא מביא אותם בדיוק ל-0.
    *   **יתרונות:**  יציבות, ביצועים טובים במקרים רבים.
    *   **חסרונות:**  לא מבצע בחירת תכונות באופן אוטומטי.
    *   **קוד (Python with Scikit-learn):**

```python
from sklearn.linear_model import Ridge
import numpy as np

# יצירת נתונים לדוגמה
X = np.random.rand(100, 10)
y = np.random.rand(100)

# אימון מודל Ridge עם מקדם רגולריזציה של 0.1
model = Ridge(alpha=0.1)
model.fit(X, y)

# הדפסת המקדמים
print(model.coef_)
```

*   **Elastic Net רגולריזציה:**
    *   **נוסחה:** שילוב של L1 ו-L2 רגולריזציה:  `λ1 * Σ |βj| + λ2 * Σ βj²`
    *   **השפעה:**  משלב את היתרונות של L1 ו-L2.  מבצע בחירת תכונות (כמו L1) ומצמצם את גודל המקדמים (כמו L2).
    *   **יתרונות:**  מתאים למקרים בהם יש הרבה תכונות קשורות.
    *   **חסרונות:**  דורש כוונון של שני מקדמי רגולריזציה (λ1 ו-λ2).
    *   **קוד (Python with Scikit-learn):**

```python
from sklearn.linear_model import ElasticNet
import numpy as np

# יצירת נתונים לדוגמה
X = np.random.rand(100, 10)
y = np.random.rand(100)

# אימון מודל ElasticNet עם מקדמי רגולריזציה של 0.1 ו-0.1
model = ElasticNet(alpha=0.1, l1_ratio=0.5)
model.fit(X, y)

# הדפסת המקדמים
print(model.coef_)
```

*   **Dropout (עבור רשתות נוירונים):**
    *   **עקרון:** במהלך האימון, באופן אקראי, חלק מהנוירונים (יחד עם הקשרים שלהם) מושבתים.  זה גורם לרשת הנוירונים ללמוד תכונות חזקות ועמידות יותר.
    *   **יתרונות:**  מניעת התאמת יתר, פשוט ליישום.
    *   **חסרונות:**  עלול להאריך את זמן האימון.
    *   **קוד (Python with Keras):**

```python
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import Dense, Dropout

model = Sequential()
model.add(Dense(128, input_dim=784, activation='relu'))
model.add(Dropout(0.2)) # 20% מהנוירונים מושבתים בכל איטרציה
model.add(Dense(10, activation='softmax'))
```

**2. בחירת מקדם הרגולריזציה (λ):**

*   **Cross-Validation:** השיטה הנפוצה ביותר לבחירת הערך האופטימלי של λ היא באמצעות Cross-Validation.  מאחזרים את הביצועים של המודל על קבוצת אימות (validation set) עבור ערכים שונים של λ. בוחרים את הערך שנותן את הביצועים הטובים ביותר.
*   **Grid Search / Random Search:**  שיטות לבחינת טווח של ערכים אפשריים עבור λ.

**3. השפעת סוג הנתונים:**

*   **נתונים רדודים (low-dimensional):** L2 רגולריזציה עשויה להיות מספקת.
*   **נתונים רדודים עם מספר רב של תכונות:** L1 רגולריזציה עשויה להיות טובה יותר לבחירת תכונות.
*   **נתונים בעלי תכונות קשורות:** Elastic Net יכולה להיות אפקטיבית.

**4. סיכום:**

רגולריזציה היא כלי חיוני בלמידת מכונה למניעת התאמת יתר ולשיפור ביצועי המודל על נתונים חדשים. הבנת סוגי הרגולריזציה השונים, היתרונות והחסרונות שלהם, וכיצד לבחור את מקדם הרגולריזציה המתאים, יכולה לשפר משמעותית את היכולת שלנו לבנות מודלים מדויקים ואמינים.
